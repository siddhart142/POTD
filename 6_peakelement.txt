class Solution {
public:
    int findPeakElement(vector<int>& nums) {
        long long n=nums.size();
        long long i;
        if(n==1)
            return 0;
        if(nums[0]>nums[1])
            return 0;
        if(nums[n-1]>nums[n-2])
            return (n-1);
        int l=0,r=n-1;
        while(l<=r)
        {
            int mid=l+((r-l)/2);
            if(nums[mid]>nums[mid+1] && nums[mid]>nums[mid-1])
                return mid;
            else if(nums[mid]>nums[mid+1])
                r=mid-1;
            else
                l=mid+1;
        }
        return r;
    }
};
